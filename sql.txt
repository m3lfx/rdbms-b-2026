Conditions
fetch all customers from nicetown

SELECT lname, addressline, town 
FROM customer 
WHERE town = 'nicetown'

fetch all customers that do not live in nicetown

SELECT lname, addressline, town 
FROM customer 
WHERE town <> 'nicetown'
ORDER BY lname DESC

FETCH ALL ITEMS that the selling price is greater than 10
SELECT description, sell_price 
FROM item
WHERE sell_price > 10
ORDER BY sell_price

fetch all orders from month of june onwards
SELECT orderinfo_id, date_placed
FROM orderinfo
WHERE  date_placed <= '2000-06-01'

fetch all customers and their orders
SELECT customer.customer_id, orderinfo.orderinfo_id, customer.fname, customer.lname, orderinfo.date_placed, orderinfo.shipping
FROM customer, orderinfo 
WHERE customer.customer_id = orderinfo.customer_id

SELECT customer.customer_id, o.orderinfo_id, c.fname, c.lname , o.date_placed, o.shipping
FROM customer c, orderinfo o 
WHERE c.customer_id = o.customer_id

fetch all customers who ordered for the month of july

SELECT c.customer_id, o.orderinfo_id, c.fname, c.lname , o.date_placed, o.shipping
FROM customer c, orderinfo o 
WHERE c.customer_id = o.customer_id
AND o.date_placed BETWEEN '2000-07-01' AND '2000-07-31'

fetch all customers and the items they bought.

SELECT o.orderinfo_id, concat(c.fname, ' ', c.lname) AS `full name`, i.item_id, i.description, ol.quantity
FROM  customer c, orderinfo o, orderline ol, item i 
WHERE c.customer_id = o.customer_id 
AND o.orderinfo_id = ol.orderinfo_id
AND i.item_id = ol.item_id
